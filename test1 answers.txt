Console.WriteLine("Enter the numebr of courses you are taking");
int courses = Convert.ToInt32(Console.ReadLine());
float tuitionFee;
const float amountPerCourse = 499f;
const float discount = .06f;
tuitionFee = courses * amountPerCourse;

if (courses > 5)
{
    float discounted = tuitionFee * discount;
    tuitionFee -= discounted;
    Console.WriteLine($"Your full tuition fee is {tuitionFee:C} with 6% discounted");
}
else
    Console.WriteLine($"Your full tuition fee is {tuitionFee:C}.");
